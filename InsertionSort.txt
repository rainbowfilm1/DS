void main()
{
int I,j,k,temp;
int a[10]= {10,9,7,101,23,44,12,78,34,23};
printf (“\n printing sorted elements...\n”);
for (k=1;k<10;k++)
{temp = a[k];
j=k-1;
while (j>=0&&temp<=a[j])
{a[j+1]= a[j];
j=j-1;
	
}

}

{printf(“\n%d\n”,a[i]);
}
